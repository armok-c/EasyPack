   ```typescript
   async function handleImport() {
     // 1. 打开文件选择器
     const selected = await open({
       multiple: false,
       title: "导入配置文件",
       filters: [{ name: "JSON", extensions: ["json"] }],
     });
     if (typeof selected !== "string") return;

     // 2. 确认弹窗（per D-23）
     const confirmed = window.confirm("确定要覆盖当前配置吗？此操作不可撤销");
     if (!confirmed) return;

     // 3. 执行导入
     await onImportProfile(selected);
   }
   ```

2. **导出流程（per D-16/D-19）：**
   ```typescript
   async function handleExport() {
     const currentName = profileMetas.find(p => p.id === activeProfileId)?.name ?? "unknown";
     const date = new Date().toISOString().split("T")[0]; // YYYY-MM-DD
     const defaultName = `easypack-${currentName}-${date}.json`;

     const selected = await save({
       defaultPath: defaultName,
       filters: [{ name: "JSON", extensions: ["json"] }],
     });
     if (typeof selected !== "string") return;

     await onExportProfile(selected);
   }
   ```

3. **添加 `save` 的 import：** `import { open, save } from "@tauri-apps/plugin-dialog"`（`open` 已存在，添加 `save`）
