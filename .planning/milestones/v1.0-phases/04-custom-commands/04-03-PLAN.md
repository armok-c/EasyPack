const deleteCommand = useCallback(async (id: string) => {
  if (commandMode === 'project' && selectedId) {
    const projectCmds = projectCommandsMap[selectedId] ?? [];
    const updated = projectCmds.filter(c => c.id !== id);

    if (updated.length === 0) {

**步骤 3：指令卡片网格渲染改造：**

替换现有的 `PRESET_COMMANDS.map(...)` 为使用 `commands` prop：

```tsx
{commands.map((cmd) => (
  <CommandCard
    key={cmd.id}
    name={cmd.name}
    icon={getIconByName(cmd.icon)}
    command={cmd.command}
    isCustom={cmd.type === 'custom'}
    editMode={editMode}
    onEdit={() => handleEdit(cmd)}
    onDelete={() => deleteCommand(cmd.id)}
    commandId={cmd.id}
    onClick={() => onExecute(cmd.command)}
  />
))}
```

**注意：** 预设指令在项目级模式下 scope='project' 但 type='preset'，仍然不可编辑/删除。全局模式下 type='preset' 的指令也不可编辑/删除 (per D-04, D-11)。只有 type='custom' 的指令才可编辑/删除。所以 isCustom 应该基于 `cmd.type === 'custom'` 而非 scope。

但在项目级模式下，预设指令（type='preset', scope='project'）应该可以删除 (per D-11: "项目级指令集中预设指令可删除")。

修正逻辑：
- isCustom（视觉标记）: `cmd.type === 'custom'`
- 可编辑/可删除: `editMode && (cmd.type === 'custom' || cmd.scope === 'project')`
- 在 CommandCard 中：editMode + (isCustom || scope === 'project') 时显示删除按钮和编辑点击
- 需要新增一个 prop 如 `editable` 来区分"可编辑"和"是自定义指令"

实际上更好的方式是：给 CommandCard 传一个 `canEdit` prop：

```typescript
<CommandCard
  ...
  isCustom={cmd.type === 'custom'}
  editMode={editMode}
  canEdit={editMode && (cmd.type === 'custom' || cmd.scope === 'project')}
  onEdit={() => handleEdit(cmd)}
  onDelete={() => deleteCommand(cmd.id)}
/>
```

CommandCard 的删除按钮和编辑点击行为由 `canEdit` 控制，而视觉标记由 `isCustom` 控制。
