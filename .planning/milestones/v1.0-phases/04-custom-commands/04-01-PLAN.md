```typescript
export interface CommandItem {
  id: string;           // crypto.randomUUID()
  name: string;         // 显示名称
  command: string;      // Shell 命令
  icon: string;         // lucide-react 图标名称字符串（如 "Terminal"）
  type: 'preset' | 'custom';
  scope: 'global' | 'project';
  addedAt: number;      // Date.now() 时间戳，排序用
}
```

**步骤 3：创建 `src/lib/icons.ts` — 图标名称映射表**

定义预设图标列表 (per D-13, UI-SPEC 图标列表) 和映射函数：

```typescript
import {
  Terminal, Code, Server, Zap, GitBranch, Package,
  Globe, Wrench, Rocket, Play,
  type LucideIcon
} from "lucide-react";

export const ICON_OPTIONS: Record<string, LucideIcon> = {
  Terminal, Code, Server, Zap, GitBranch, Package,
  Globe, Wrench, Rocket, Play,
};

export const DEFAULT_ICON = "Terminal";

export function getIconByName(name: string): LucideIcon {
  return ICON_OPTIONS[name] ?? Terminal;
}
```

10 个图标名称字符串：Terminal, Code, Server, Zap, GitBranch, Package, Globe, Wrench, Rocket, Play。

**步骤 4：扩展 `src/lib/presets.ts` — 添加预设到 CommandItem 转换**

保留现有 `PresetCommand` 接口和 `PRESET_COMMANDS` 数组不变（现有代码依赖它们）。新增转换函数：

```typescript
import { CommandItem } from './types';

// 将预设指令转换为 CommandItem 格式（用于全局模式展示）
export function getPresetAsCommandItems(): CommandItem[] {
  return PRESET_COMMANDS.map((cmd, idx) => ({
    id: `preset-${idx}`,
    name: cmd.name,
    command: cmd.command,
    icon: cmd.icon.displayName ?? 'Package',
    type: 'preset' as const,
    scope: 'global' as const,
    addedAt: idx,
  }));
}
```

注意：LucideIcon 的 displayName 不可靠（RESEARCH.md Open Question 1），所以对预设指令直接硬编码图标名称更安全。改为手动指定：

```typescript
const PRESET_ICON_NAMES = ['Package', 'Play', 'GitBranch', 'Sparkles'] as const;

export function getPresetAsCommandItems(): CommandItem[] {
  return PRESET_COMMANDS.map((cmd, idx) => ({
    id: `preset-${idx}`,
    name: cmd.name,
    command: cmd.command,
    icon: PRESET_ICON_NAMES[idx],
    type: 'preset' as const,
    scope: 'global' as const,
    addedAt: idx,
  }));
}
```
